# Horde vs Civilized
cb_horde_vs_civ = {
	valid_for_subject = no

	prerequisites = {
		is_neighbor_of = FROM
		FROM = {
			NOT = {
				is_nomad = yes
				has_government_attribute = non_native_tribal
			}
		}
		has_government_attribute = non_native_tribal
		is_revolutionary = no
	}

	war_goal = superiority_horde
}

# Burn European Colonies
cb_push_back_colonizers = {
	valid_for_subject = no

	prerequisites = {
		has_dlc = "Conquest of Paradise"
		is_native_tribe = yes
		NOT = { has_government_attribute = non_native_tribal }
		FROM = {
			is_native_tribe = no
			OR = {
				is_neighbor_of = ROOT
				any_owned_province = {
					tribal_land_of = ROOT
				}
			}
		}
	}

	war_goal = push_back_colonizers
}
