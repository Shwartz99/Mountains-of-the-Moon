estate_slaves_atlantic_slave_trade = {
	icon = privilege_atlantic_slave_trade
	loyalty = -0.1
	influence = 0.25
	is_valid = {
		OR = {
			any_owned_province = {
				OR = {
					continent = north_america
					continent = south_america
					continent = oceania
					continent = new_world
				}
			}
			any_subject_country = {
				is_colonial_nation = yes
				any_owned_province = {
					OR = {
						continent = north_america
						continent = south_america
						continent = oceania
						continent = new_world
					}
				}
			}
		}
		any_owned_province = {
			trade_goods = slaves
			continent = africa
		}
	}
	can_select = {
		capital_scope = {
			continent = europe
		}
		OR = {
			any_owned_province = {
				OR = {
					continent = north_america
					continent = south_america
					continent = oceania
					continent = new_world
				}
			}
			any_subject_country = {
				is_colonial_nation = yes
				any_owned_province = {
					OR = {
						continent = north_america
						continent = south_america
						continent = oceania
						continent = new_world
					}
				}
			}
		}
		any_owned_province = {
			trade_goods = slaves
			continent = africa
		}
	}
	on_granted = {
		every_owned_province = {
			limit = {
				trade_goods = slaves
				continent = africa
			}
			add_province_modifier = {
				name = atlantic_slave_export_modifier
				duration = -1
			}
		}
		every_owned_province = {
			limit = {
				OR = {
					trade_goods = coffee
					trade_goods = cotton
					trade_goods = sugar
					trade_goods = tobacco
					trade_goods = tea
					trade_goods = cocoa
				}
				OR = {
					continent = north_america
					continent = south_america
					continent = oceania
					continent = new_world
				}
			}
			add_province_modifier = {
				name = atlantic_slave_import_modifier
				duration = -1
			}
		}
		every_subject_country = {
			limit = { is_colonial_nation = yes }
			every_owned_province = {
				limit = {
					OR = {
						trade_goods = coffee
						trade_goods = cotton
						trade_goods = sugar
						trade_goods = tobacco
						trade_goods = tea
						trade_goods = cocoa
					}
					OR = {
						continent = north_america
						continent = south_america
						continent = oceania
						continent = new_world
					}
				}
				add_province_modifier = {
					name = atlantic_slave_import_modifier
					duration = -1
				}
			}
		}
	}
	cooldown_years = 50
	on_revoked = {
		every_owned_province = {
			limit = { OR = { has_province_modifier = atlantic_slave_import_modifier has_province_modifier = atlantic_slave_export_modifier} }
			remove_province_modifier = atlantic_slave_import_modifier
			remove_province_modifier = atlantic_slave_export_modifier
		}
		every_subject_country = {
			limit = { is_colonial_nation = yes }
			every_owned_province = {
				limit = { OR = { has_province_modifier = atlantic_slave_import_modifier has_province_modifier = atlantic_slave_export_modifier} }
				remove_province_modifier = atlantic_slave_import_modifier
				remove_province_modifier = atlantic_slave_export_modifier
			}
		}
	}
	on_invalid = {
	}
	penalties = {
	}
	benefits = {
	}
	ai_will_do = {
		factor = 25
	}
}
