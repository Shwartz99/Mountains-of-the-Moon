namespace = motm

#Culture change to sudanese
province_event = {
	id = motm.1
	title = motm.1.t
	desc = motm.1.d
	picture = CULTURE_eventPicture
	goto = root

	trigger = {
		owner = {
			OR = {
				primary_culture = sudanese_culture
				culture_group = sudanese
			}
		}
		culture_group = sudanese
		any_neighbor_province = {
			culture_group = turko_semitic
		}
	}

	is_mtth_scaled_to_size = 1

	mean_time_to_happen = {
		months = 2400
		modifier = {
			factor = 0.8
			owner = {
				accepted_culture = sudanese_culture
			}
		}
		modifier = {
			factor = 0.9
			owner = {
				OR = {
					any_owned_province = {
						culture = sudanese_culture
					}
					religion_group = muslim
				}
			}
		}
		modifier = {
			factor = 0.5
			owner = { primary_culture = sudanese_culture }
		}
	}

	option = {
		name = motm.1.a
		change_culture = sudanese_culture
		add_base_tax = 1
		ai_chance = {
			factor = 1
		}
	}
	option = {
		name = motm.1.b
		add_province_modifier = {
			name = suppress_minority
			duration = 3650
		}
		ai_chance = {
			factor = 0
		}
	}
}

#Improving the fortifications

province_event = {
	id = motm.3
	title = "motm.3.t"
	desc = "motm.3.d"
	picture = FORT_eventPicture
	
	trigger = {
	NOT={has_province_modifier = hausa_fort_modifier}
	development=10
	is_capital = yes
	culture=hausa
	
	owner = { 
	NOT={is_at_war=yes}
	primary_culture=hausa
	NOT={development=60}
	treasury = 100 
	}
	}

	mean_time_to_happen = {
	months = 120 	
	}
	
	option = {
		name=motm.3.a #Improve The Fortifications
		add_province_modifier = { 
		name = hausa_fort_modifier
		duration = -1  
		}
		owner={
		add_treasury=-50
		add_mil_power=50
		}
		
	}
	
	option = {
		name=motm.3.b #Leave them as they are
		owner={
		add_adm_power=50
		add_mil_power=-50
		}
	}
	
}

#Debate over alliance with Pagans

country_event = {
	id = motm.4
	title = "motm.4.t"
	desc = "motm.4.d"
	picture = RELIGIOUS_TURMOIL_eventPicture
	
	trigger = {
	any_ally = {
	religion=shamanism
	}
	ROOT={religion_group=muslim}
	}

	mean_time_to_happen = {
	months = 0 	
	}
	
	option = {
		name= motm.4.a #Change policy
		every_ally = {
		limit={religion=shamanism}
		reverse_add_opinion = {
		who = ROOT	
		modifier = intolerant_with_pagans	
		}
		add_opinion = {
		who = ROOT	
		modifier = intolerant_with_pagans	
		}
		}
		add_estate_loyalty = {
		estate = estate_church
		loyalty = 15
		}
		add_estate_influence_modifier = {
		estate = estate_church
		desc = EST_VAL_INTOLERANCE
		influence = 10
		duration = 3650
		}
	}
		
	
	option = {
		name=motm.4.b #Maintain policy
		add_estate_loyalty = {
		estate = estate_church
		loyalty = -15
		}
		add_estate_influence_modifier = {
		estate = estate_church
		desc = EST_VAL_TOLERANCE
		influence = -10
		duration = 3650
		}
	}
	
}
